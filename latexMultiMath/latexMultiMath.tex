%导言区
\documentclass{article}
\usepackage{ctex}
\usepackage{amsmath}
\usepackage{amssymb}

%正文区
\begin{document}
	%gather和gather*环境（可以使用\\换行）
	%带编号
	\begin{gather} %有编号
	a + b =b + a\\
	ab ba
	\end{gather}
	 
	\begin{gather*} %无编号
	3 + 5 = 5 + 3 =8 \\
	3 \times 5 = 5 \times 3
	\end{gather*}
	
	%在\\前使用\notag阻止编号
	\begin{gather} %有编号
		3 + 1 = 4 \notag\\
		5 + 6 =11 \notag\\
		7 + 9 = 16 \notag
		\end{gather}
	
	
	% align 和 align*环境（用 & 进行对齐）
	% 带编号
	\begin{align}
	x &= t +\cos t + 1\\
	y &= 2\sin t
	\end{align}
	
	%不带编号
	\begin{align*}
	x &= t & x &= \cos t & x & =t\\
	y &= 2t &  y &=\sin (t+1) & y &=\sin t
	\end{align*}
	
	%split环境（对齐采用align环境的方式，编号在中间）
	\begin{equation}
	\begin{split}
	\cos 2x &= \cos^2 x - \sin^2 x\\
	&= 2\cos^2 x - 1
	\end{split}
	\end{equation}
	
	% cases 环境
	% 每行公式中使用 & 分隔为两部分
	% 通常表示值和后面的条件
	%mathbb：输出花体字符，需要amssymb宏包支持；\in：输出属于符号 \text命令用于数学模式中转换成文本模式，输出文字，否则无法输出中文
	\begin{equation}
	D(x) = \begin{cases}
	1, & \text{如果 } x  \in \mathbb{Q};\\
	0, & \text{如果 } x  \in 
	\mathbb{R}\setminus\mathbb{Q}
	\end{cases}
	\end{equation}
	
\end{document}
