%导言区
\documentclass{article}

\usepackage{ctex}%中文处理宏包
\usepackage{xltxtra}%此宏包是为了得到XeLaTeX标志符
\usepackage{texnames} %产生其他的标志符,一些logo

%正文区（文稿区）
\begin{document}
	\section{空白符号}
	%空行分段，多个空行等同1个
	%自动缩进，绝对不能使用空格代替
	%英文中多个空格等效于一个空格，中文中的空格将会被忽略
	%汉字与其他字符的间距会自动有XeLaTex处理
	
	%1em(当前字体中M的宽度,一个字宽)
	a\quad b
	
	%2em
	a\qquad b
	
	%约为1/6个em
	a\,b  a\thinspace b
	
	%0.5个em
	a\enspace b
	
	%空格
	a\ b
	
	%硬空格（不能分割的空格）
	a~b
	
	%产生指定宽度的空白
	%1pc=12pt=4.218mm
	a\kern 1pc b
	
	a\kern -1em b
	
	a\hskip 1em b
	
	a\hspace{35pt}b
	
	%占位宽度
	a\hphantom{xyz}b
	
	%弹性长度
	a\hfill b
	
	
	\section{\LaTeX 控制符}
	\# \$ \% \{ \} \~{} \_{} \^{} \textbackslash \&
	
	\section{排版符号}
	\S \P \dag \ddag \copyright \pounds
	
	\section{Tex 标志符号}
	%基本符号
	Tex{} %此标志符号不能正常显示
	\LaTeX{} 
	\LaTeXe{}
	%xltxtra宏包提供
	\XeLaTeX
	
	
	\section{引号}
	` ' `` '' \\
	``伍慧兰，你好可爱哦''
	
	\section{连字符}
	- -- ---
	
	\section{非英文字母}
	\oe\  \OE\  \ae\ \AE~ \aa ~ \AA~ \o ~\O ~\l~ \L ~!`~ ?`
	
	\section{重音符号(以o为例)}
	\`o  \'o   \^o  \''o \~o \=o \.o \u{o} \v{o} 
	\H{o} \r{o} \t{o} \b{o} \c{o} \d{o}
	
\end{document}
